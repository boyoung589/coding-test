function solution(people, limit) {
    let boat = 0;
    people.sort((a, b)=> b-a);
    let first = 0;
    let last = people.length-1;
    while(last>=first){
        if(last === first){
            return ++boat;
        } 
        if(people[first]+people[last]<=limit){
            boat++;
            first++;
            last--;
        } else{
            boat++;
            first++;
        }
    }
    return boat;
}
