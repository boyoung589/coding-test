function solution(s) {
    let answer = [];
    let devided = s.split(' ');
    for(let word of devided){
        answer.push(word.split('').map((el, idx)=>{
            return idx%2 === 0 ? el.toUpperCase() : el.toLowerCase();
        }).join(''))
    }
    return answer.join(' ');
}
